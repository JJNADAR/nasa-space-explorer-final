const API_KEY = "DEMO_KEY"; // works for class projects const gallery = document.getElementById("gallery"); const startDateInput = document.getElementById("startDate"); const endDateInput = document.getElementById("endDate"); const button = document.querySelector("button"); // Loading message function showLoading() { gallery.innerHTML = `

Loading space images...

`; } // Random space fact (LevelUp) function showFact() { const facts = [ "One day on Venus is longer than one year on Venus.", "Neutron stars can spin 600 times per second.", "There are more stars in the universe than grains of sand on Earth.", "A spoonful of a neutron star would weigh billions of tons." ]; const fact = facts[Math.floor(Math.random() * facts.length)]; const factBox = document.createElement("div"); factBox.className = "fact-box"; factBox.innerHTML = `Did You Know?

${fact}

`; gallery.prepend(factBox); } // Fetch NASA data async function fetchNASA(start, end) { showLoading(); const url = `https://api.nasa.gov/planetary/apod?api_key=${API_KEY}&start_date=${start}&end_date=${end}`; const res = await fetch(url); const data = await res.json(); renderGallery(data.slice(0, 9)); // rubric requirement showFact(); } // Render gallery function renderGallery(items) { gallery.innerHTML = ""; items.forEach((item) => { const div = document.createElement("div"); div.className = "card"; div.innerHTML = `

${item.title}

${item.date}

`; gallery.appendChild(div); }); } // Modal function openModal(item) { const modal = document.createElement("div"); modal.className = "modal"; modal.innerHTML = `

${item.title}

${item.date}

${item.explanation}

`; document.body.appendChild(modal); } // Button click button.addEventListener("click", () => { const start = startDateInput.value; const end = endDateInput.value; if (!start || !end) { alert("Please select both dates"); return; } fetchNASA(start, end); });
